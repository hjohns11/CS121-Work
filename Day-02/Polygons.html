drawCircle <- function(x, y, r) {
    angs <- seq(0, 2 * pi, length = 400)
    xpts <- x + r * cos(angs)
    ypts <- y + r * sin(angs)
    polygon(xpts, ypts, col = "light blue", density = 100)
}
drawCircle(50, 50, 25)

drawEllipse <- function(x, y, r) {
    angs <- seq(0, 2 * pi, length = 400)
    xpts <- x + r * cos(angs)
    ypts <- y + 25 * sin(angs)
    polygon(xpts, ypts, col = "purple", density = 80)
}
drawEllipse(22.5, 70, 15)
